001package pkg_game.pkg_room;
002
003import pkg_game.Item;
004
005import java.io.Serializable;
006
007public class Door implements Serializable {
008    private Room front;
009    private Room back;
010
011    private boolean locked;
012    private Item key;
013
014    public Door(Room front, Room back) {
015        this.front = front;
016        this.back = back;
017    }
018
019    public Room getFront() {
020        return front;
021    }
022
023    public Room getBack() {
024        return back;
025    }
026
027    public boolean isLocked() {
028        return locked;
029    }
030
031    public void setLocked(boolean locked) {
032        this.locked = locked;
033    }
034
035    public Item getKey() {
036        return key;
037    }
038
039    public void setKey(Item key) {
040        this.key = key;
041    }
042
043    /**
044     * Get the room at the other side of this door
045     * @param from Where you are standing
046     * @return A room, or null
047     */
048    public Room getOtherSide(final Room from){
049        if(from.equals(front))return back;
050        if(from.equals(back))return front;
051        return null;
052    }
053
054}

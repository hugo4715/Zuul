001package pkg_game;
002
003import java.io.Serializable;
004
005public class Item implements Serializable {
006    private String name;
007    private String description;
008    private int weight;
009
010    public Item(String name, final String description, final int weight) {
011        this.name = name;
012        this.description = description;
013        this.weight = weight;
014    }
015
016    public String getName() {
017        return name;
018    }
019
020    public int getWeight() {
021        return weight;
022    }
023
024    public String getDescription() {
025        return description;
026    }
027
028    public String getLongDescription() {
029        return getName() + "( " + getDescription() + ")";
030    }
031}

001package pkg_game.pkg_command;
002
003import pkg_game.GameEngine;
004import pkg_game.Player;
005import pkg_game.pkg_room.Room;
006
007public class CommandBack extends Command {
008    protected CommandBack(GameEngine engine) {
009        super(engine);
010    }
011
012    @Override
013    public void execute(Player player) {
014        if (!player.goBack()) {
015            gui.println("You can't go back");
016        } else {
017            Room currentRoom = player.getCurrentRoom();
018            engine.tickEntities();
019            gui.println(currentRoom.getLongDescription());
020            if (currentRoom.getImageName() != null)
021                gui.showImage(currentRoom.getImageName());
022        }
023    }
024}

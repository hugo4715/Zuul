001package pkg_game.pkg_command;
002
003import pkg_game.GameEngine;
004import pkg_game.UserInterface;
005
006import java.io.Serializable;
007import java.util.HashMap;
008import java.util.Map;
009
010public class CommandWords implements Serializable {
011    private Map<String,Command> commands;
012
013    public CommandWords(final GameEngine engine) {
014        this.commands = new HashMap<>();
015
016        commands.put("go", new CommandGo(engine));
017        commands.put("alea",new CommandAlea(engine));
018        commands.put("back", new CommandBack(engine));
019        commands.put("drop", new CommandDrop(engine));
020        commands.put("eat", new CommandEat(engine));
021        commands.put("fire", new CommandFire(engine));
022        commands.put("help", new CommandHelp(engine));
023        commands.put("items", new CommandItems(engine));
024        commands.put("load", new CommandLoad(engine));
025        commands.put("look", new CommandLook(engine));
026        commands.put("quit", new CommandQuit(engine));
027        commands.put("save", new CommandSave(engine));
028        commands.put("take", new CommandTake(engine));
029        commands.put("test", new CommandTest(engine));
030        commands.put("use", new CommandUse(engine));
031        commands.put("talk", new CommandTalk(engine));
032    }
033
034    public void setGui(final UserInterface userInterface){
035        commands.values().forEach(cmd -> cmd.setGui(userInterface));
036    }
037
038    /**
039     * Get all the available commands
040     *
041     * @return a String with all the command available
042     */
043    public String getCommandList() {
044        StringBuilder msg = new StringBuilder();
045        for (String vCmd : commands.keySet()) {
046            msg.append(vCmd);
047            msg.append(' ');
048        }
049        msg.append('\n');
050        return msg.toString();
051    }
052
053    /**
054     * Verifie si une String donnee fait partie des commandes valides.
055     *
056     * @param command la String a tester
057     * @return true si pString est une comande valide, false sinon
058     */
059    public boolean isCommand(final String command) {
060        return commands.containsKey(command);
061    }
062
063
064    public Command getCommand(String vWord1) {
065        return commands.get(vWord1);
066    }
067
068}

001package pkg_game.pkg_entity;
002
003import pkg_game.pkg_room.Door;
004import pkg_game.pkg_room.Room;
005
006import java.io.Serializable;
007
008public class Frog implements Serializable,Entity, EntityTalkable, EntityMoving {
009    private String name;
010    private String text;
011
012    private Room room;
013
014    public Frog(Room room, String name, String text) {
015        this.name = name;
016        this.text = text;
017        this.room = room;
018    }
019
020    public String getName() {
021        return name;
022    }
023
024    public String getText() {
025        return text;
026    }
027
028    @Override
029    public Room getRoom() {
030        return room;
031    }
032
033    @Override
034    public void move() {
035        Door door = room.getRandomExit();
036        if(door != null && !door.isLocked()) {
037            room = door.getOtherSide(room);
038            System.out.println(getName() + "moved to " + room.getName());//debug
039        }
040    }
041
042    @Override
043    public double moveChance() {
044        return 0.25;
045    }
046}

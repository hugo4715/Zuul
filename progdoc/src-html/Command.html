002/**
003 * A class representing a command
004 */
005public class Command {
006    private String commandWord;
007    private String secondWord;
008
009    /**
010     * Create a new command
011     *
012     * @param commandWord The first command word
013     * @param secondWord  The second command word
014     */
015    public Command(final String commandWord, final String secondWord) {
016        this.commandWord = commandWord;
017        this.secondWord = secondWord;
018    }
019
020    /**
021     * Get the first command word
022     *
023     * @return The first Command word
024     */
025    public String getCommandWord() {
026        return this.commandWord;
027    }
028
029    /**
030     * Get the second command word
031     *
032     * @return The second Command word
033     */
034    public String getSecondWord() {
035        return this.secondWord;
036    }
037
038    /**
039     * Get if the command has a second word
040     *
041     * @return true if the command has a second word
042     */
043    public boolean hasSecondWord() {
044        return this.secondWord != null;
045    }
046
047    /**
048     * Get if the command is unknown
049     *
050     * @return true if the command is unknown (no first word)
051     */
052    public boolean isUnknown() {
053        return this.commandWord == null;
054    }
055}

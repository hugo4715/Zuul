001import javax.swing.*;
002import java.awt.*;
003import java.awt.event.*;
004import java.net.URL;
005
006/**
007 * This class implements a simple graphical user interface with a text entry
008 * area, a text output area and an optional image.
009 *
010 * @author Michael Kolling (DB edited)
011 * @version 1.0 (Jan 2003)
012 */
013public class UserInterface implements ActionListener {
014    private GameEngine engine;
015    private JFrame frame;
016    private JTextField entryField;
017    private JTextArea log;
018    private JLabel image;
019
020    private JButton buttonLook;
021    private JButton buttonnEat;
022    private JButton buttonQuit;
023
024    /**
025     * Construct a UserInterface. As a parameter, a Game Engine
026     * (an object processing and executing the game commands) is
027     * needed.
028     *
029     * @param gameEngine The GameEngine object implementing the game logic.
030     */
031    public UserInterface(final GameEngine gameEngine) {
032        this.engine = gameEngine;
033        this.createGUI();
034    }
035
036    /**
037     * Print out some text into the text area.
038     */
039    public void print(final String text) {
040        this.log.append(text);
041        this.log.setCaretPosition(this.log.getDocument().getLength());
042    }
043
044    /**
045     * Print out some text into the text area, followed by a line break.
046     */
047    public void println(final String text) {
048        this.print(text + "\n");
049    } // println(.)
050
051    /**
052     * Show an image file in the interface.
053     */
054    public void showImage(final String imageName) {
055        URL vImageURL = this.getClass().getClassLoader().getResource(imageName);
056        if (vImageURL == null)
057            System.out.println("image not found");
058        else {
059            ImageIcon vIcon = new ImageIcon(vImageURL);
060            this.image.setIcon(vIcon);
061            this.frame.pack();
062        }
063    }
064
065    /**
066     * Enable or disable input in the input field.
067     */
068    public void setInputEnabled(final boolean enabled) {
069        this.entryField.setEditable(enabled);
070        if (!enabled)
071            this.entryField.getCaret().setBlinkRate(0);
072    }
073
074    /**
075     * Set up graphical user interface.
076     */
077    private void createGUI() {
078        this.frame = new JFrame("World of Zuul");
079        this.entryField = new JTextField(34);
080
081        this.log = new JTextArea();
082        this.log.setEditable(false);
083        JScrollPane vListScroller = new JScrollPane(this.log);
084        vListScroller.setPreferredSize(new Dimension(200, 200));
085        vListScroller.setMinimumSize(new Dimension(100, 100));
086
087        JPanel vPanel = new JPanel();
088        this.image = new JLabel();
089
090
091        this.buttonLook = new JButton("Look");
092        this.buttonnEat = new JButton("Eat");
093        this.buttonQuit = new JButton("Quit");
094
095        JPanel buttonPanel = new JPanel();
096        buttonPanel.setLayout(new BoxLayout(buttonPanel, BoxLayout.PAGE_AXIS));
097        buttonPanel.add(this.buttonLook);
098        buttonPanel.add(this.buttonnEat);
099        buttonPanel.add(this.buttonQuit);
100
101        vPanel.setLayout(new BorderLayout());
102        vPanel.add(this.image, BorderLayout.NORTH);
103        vPanel.add(vListScroller, BorderLayout.CENTER);
104        vPanel.add(this.entryField, BorderLayout.SOUTH);
105        vPanel.add(buttonPanel, BorderLayout.EAST);
106
107        this.frame.getContentPane().add(vPanel, BorderLayout.CENTER);
108
109        // add some event listeners to some components
110        this.frame.addWindowListener(new WindowAdapter() {
111            public void windowClosing(final WindowEvent event) {
112                System.exit(0);
113            }
114        });
115
116        this.buttonLook.addActionListener(this);
117        this.entryField.addActionListener(this);
118        this.buttonQuit.addActionListener(this);
119        this.buttonnEat.addActionListener(this);
120
121        this.frame.pack();
122        this.frame.setVisible(true);
123        this.entryField.requestFocus();
124    }
125
126    /**
127     * Actionlistener interface for entry textfield.
128     */
129    public void actionPerformed(final ActionEvent event) {
130
131        if (event.getSource() == this.buttonQuit) {
132            engine.interpretCommand("quit");
133        } else if (event.getSource() == this.entryField) {
134            this.processCommand();
135        }else if(event.getSource() == this.buttonLook){
136            engine.interpretCommand("look");
137        }else if(event.getSource() == this.buttonnEat){
138            engine.interpretCommand("eat");
139        }
140
141    }
142
143    /**
144     * A command has been entered. Read the command and do whatever is
145     * necessary to process it.
146     */
147    private void processCommand() {
148        String vInput = this.entryField.getText();
149        this.entryField.setText("");
150
151        this.engine.interpretCommand(vInput);
152    }
153}

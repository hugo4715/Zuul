001import java.util.HashMap;
002import java.util.Map;
003import java.util.Stack;
004
005/**
006 *  This class is part of the "World of Zuul" application. 
007 *  "World of Zuul" is a very simple, text based adventure game.
008 * 
009 *  This class creates all rooms, creates the parser and starts
010 *  the game.  It also evaluates and executes the commands that 
011 *  the parser returns.
012 * 
013 * @author  Michael Kolling and David J. Barnes
014 * @version 1.0 (Jan 2003)
015 */
016public class GameEngine {
017
018    private Parser parser;
019    private Room currentRoom;
020    private Stack<Room> lastRooms;
021    private UserInterface gui;
022    private Map<String,Room> rooms;
023
024    /**
025     * Constructor for objects of class GameEngine
026     */
027    public GameEngine(){
028        this.rooms = new HashMap<>();
029        parser = new Parser();
030        this.lastRooms = new Stack<>();
031        createRooms();
032    }
033
034    public void setGUI(UserInterface userInterface){
035        gui = userInterface;
036        printWelcome();
037    }
038
039    /**
040     * Print out the opening message for the player.
041     */
042    private void printWelcome(){
043        gui.print("\n");
044        gui.println("Welcome to the World of Zuul!");
045        gui.println("World of Zuul is a new, incredibly boring adventure game.");
046        gui.println("Type 'help' if you need help.");
047        gui.print("\n");
048        gui.println(currentRoom.getLongDescription());
049        gui.showImage(currentRoom.getImageName());
050    }
051
052    /**
053     * Create all the rooms and link their exits together.
054     */
055    private void createRooms(){
056        String vDefaultImage = "default.png";
057
058
059        Item itemBattery = new Item("battery",1);
060        Item itemScrewdriver = new Item("screwdriver",1);
061
062        Room vPrison = new Room("locked inside a small prison cell.\nThe power just went off and the door in front off you just openned, you can now get out of this cell. ",vDefaultImage);
063        vPrison.addItem(itemBattery);
064        vPrison.addItem(itemScrewdriver);
065
066        this.rooms.put("prison",vPrison);
067
068        Room vCorridor1 = new Room("now inside a small corridor. \nYou can see a two doors, but there are some heavy creates in front of one, you will need to find something to move them. ",vDefaultImage);
069        this.rooms.put("corridor1",vCorridor1);
070
071        Room vCorridor2 = new Room("now in another corridor, you can see two closed doors.",vDefaultImage);
072        this.rooms.put("corridor2",vCorridor2);
073
074        Room vCafeteria = new Room("in inside what look like a cafeteria, you can get some food supply here, \nbut right now, that's not your priority.",vDefaultImage);
075        this.rooms.put("cafetaria",vCafeteria);
076
077        Room vLabo = new Room("now inside a laboratory. \nIt's empty but you can see some strange animals floating inside an aquarium. You can see on a table something lighting up. From the papers on the desks, it's a gravity gun. ",vDefaultImage);
078        this.rooms.put("labo",vLabo);
079
080        vPrison.setExit("east", vCorridor1);
081        vCorridor1.setExit("west",vPrison);
082
083        vCorridor1.setExit("south", vCorridor2);
084        vCorridor2.setExit("north", vCorridor1);
085
086        vCorridor2.setExit("east", vCafeteria);
087        vCafeteria.setExit("west", vCorridor2);
088
089        vLabo.setExit("down",vCafeteria);
090        vCafeteria.setExit("up", vLabo);
091
092        this.currentRoom = vPrison;
093    }
094
095    /**
096     * Given a command, process (that is: execute) the command.
097     * If this command ends the game, true is returned, otherwise false is
098     * returned.
099     */
100    public void interpretCommand(final String commandLine) {
101        gui.println(commandLine);
102        Command command = parser.getCommand(commandLine);
103
104        if(command.isUnknown()) {
105            gui.println("I don't know what you mean...");
106            return;
107        }
108
109        switch (command.getCommandWord()){
110            case "help":
111                printHelp();
112                break;
113            case "go":
114                goRoom(command);
115                break;
116            case "quit":
117                if(command.hasSecondWord()) {
118                    gui.println("Quit what?");
119                } else {
120                    endGame();
121                }
122                break;
123            case "eat":
124                eat(command);
125                break;
126            case "look":
127                look(command);
128                break;
129            case "back":
130                goBack(command);
131                break;
132            default:
133                gui.println("I don't know what you mean...");
134
135        }
136    }
137
138    // implementations of user commands:
139
140    private void goBack(Command command) {
141        if(lastRooms.isEmpty()){
142            gui.println("There is no room to go back to!");
143        }else{
144            currentRoom = lastRooms.pop();
145            gui.println(currentRoom.getLongDescription());
146            if(currentRoom.getImageName() != null)
147                gui.showImage(currentRoom.getImageName());
148        }
149
150    }
151    /**
152     * Print out some help information.
153     * Here we print some stupid, cryptic message and a list of the 
154     * command words.
155     */
156    private void printHelp() {
157        gui.println("You are lost. You are alone in the spaceship");
158        gui.print("Your command words are: " + parser.getCommands());
159    }
160
161    /** 
162     * Try to go to one direction. If there is an exit, enter the new
163     * room, otherwise print an error message.
164     */
165    private void goRoom(final Command command) {
166        if(!command.hasSecondWord()) {
167            // if there is no second word, we don't know where to go...
168            gui.println("Go where?");
169            return;
170        }
171
172        String direction = command.getSecondWord();
173
174        // Try to leave current room.
175        Room nextRoom = currentRoom.getExit(direction);
176
177        if (nextRoom == null)
178            gui.println("There is no door!");
179        else {
180            lastRooms.push(currentRoom);
181            currentRoom = nextRoom;
182            gui.println(currentRoom.getLongDescription());
183            if(currentRoom.getImageName() != null)
184                gui.showImage(currentRoom.getImageName());
185        }
186    }
187
188    private void printLocationInfo(){
189        this.gui.println(this.currentRoom.getLongDescription());
190    }
191
192    /**
193     * Handle the look command
194     * @param command The Command to handle
195     */
196    private void look(final Command command){
197        this.printLocationInfo();
198    }
199
200    /**
201     * Handle the eat command
202     * @param command Handle the eat command
203     */
204    private void eat(final Command command){
205        this.gui.println("You have eaten now and you are not hungry any more.");
206    }
207
208
209    private void endGame(){
210        gui.println("Thank you for playing.  Good bye.");
211        gui.setInputEnabled(false);
212    }
213
214}

001
002import java.util.HashMap;
003import java.util.HashSet;
004import java.util.Set;
005
006/**
007 * A class representing a room
008 */
009public class Room {
010    private String description;
011    private HashMap<String, Room> exits;
012    private String imageName;
013    private Set<Item> items;
014
015    /**
016     * Create a new room
017     *
018     * @param description The room description
019     * @param image The room image
020     */
021    public Room(final String description, final String image) {
022        this.description = description;
023        this.exits = new HashMap<>();
024        this.imageName = image;
025        this.items = new HashSet<>();
026    }
027
028    /**
029     * Change the exit with the specified name/direction
030     *
031     * @param direction The direction of the exit
032     * @param room      The room which the exit is pointing to
033     */
034    public void setExit(final String direction, final Room room) {
035        this.exits.put(direction, room);
036    }
037
038    /**
039     * Get the room description
040     *
041     * @return The room description
042     */
043    public String getDescription() {
044        return this.description;
045    }
046
047    /**
048     * Get the exit with the specified name
049     *
050     * @param exitName The exit name
051     * @return A Room, of null if there isn't any exit with the specified name
052     */
053    public Room getExit(String exitName) {
054        return this.exits.get(exitName);
055    }
056
057    /**
058     * Get a string representing all the available exits in the room
059     *
060     * @return A String with the exits
061     */
062    public String getExitString() {
063        StringBuilder vString = new StringBuilder("Exits: ");
064        for (String key : this.exits.keySet()) {
065            vString.append(key);
066            vString.append(' ');
067        }
068        return vString.toString();
069    }
070
071    public String getImageName() {
072        return imageName;
073    }
074
075    /**
076     * Return the long description of the room
077     *
078     * @return The long description of this room (eg. 'You are in [...]. Exits: [...]
079     */
080    public String getLongDescription() {
081        String msg = "You are " + this.getDescription()
082                + "\n" + this.getExitString();
083        if(this.items.isEmpty()) {
084            msg += "\nNo item here";
085        }else{
086            msg += "\nItems: ";
087            for(Item item : this.items){
088                msg += "\n  - " + item.getDescription();
089            }
090        }
091        return msg;
092    }
093
094    public void addItem(Item item) {
095        this.items.add(item);
096    }
097
098    public Set<Item> getItems() {
099        return items;
100    }
101}
